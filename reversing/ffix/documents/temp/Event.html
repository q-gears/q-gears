// ===========================================
// Final Fantasy IX | Event | Header
// Size: 128 bytes
// ===========================================
00 | 45 56       = "EV"      // Event Block ID
02 | 02          = 2         // Alpha Count
03 | 0b          = 11        // Beta Count

// ===========================================
// Final Fantasy IX | Event | Alpha
// Size: 8 bytes
// ===========================================
00 | 58 00       = 88        // Event Address
02 | 0c 00       = 12        // Event Size
04 | 00 00       = 0         // ?
06 | 00 00       = 0         // ?

// ===========================================
// Final Fantasy IX | Event | Beta
// Size: 8 bytes
// ===========================================
00 | 58 00       = 88        // Event Address
02 | 0c 00       = 12        // Event Size
04 | 00 00       = 0         // ?
06 | 00 00       = 0         // ?
